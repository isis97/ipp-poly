51 # Polynomial library
52 
53 All the documentation is provided in header files.
54 
55 There you've got example usage of the library:
56 
57 ```c
58  #include "poly.h"
59 
60  int main(void) {
61 
62  Poly p = PolyC(4);
63  // Create constant polynomial
64  // p(x) = 4
65 
66  Poly q = PolyP( PolyC(1), 2 );
67  // Create polynomial that consists of
68  // variable square multiplied by
69  // coefficient 1
70  // q(x) = 1*x^2
71 
72  // As you may see the coefficients that are placed
73  // next to variables may be other polynomials
74  // (polynomials from other variable)
75 
76  Poly r = PolyP( PolyP( PolyC(1), 2 ), 3 );
77  // In this example
78  // r(x,y) = (1*y^2)*(x^3)
79 
80  // Now you can do arithemtic operations on them
81  // for example addition:
82  Poly sum = PolyAdd(&p, &q);
83  // sum(x) = 4 + x^2
84  // If you are not sure what would happen if we sum
85  // q and r then we say that the result is
86  //
87  // q(x) + r(x,y) = 1*x^2 + (1*y^2)*(x^3)
88  // that is:
89  // x^2 + (y^2)(x^3)
90  //
91 
92  // Or evalue at specific point:
93  Poly rAtTwo = PolyAt(&r, 2);
94  // Now we assume the first variable is equal to 2
95  // So we calculate r(x,y) for x->2
96  // rAtTwo = r(2,y) = (2^3)*y^2 = 8*y^2
97 
98  // You can print them nicely:
99  PolyPrint(&rAtTwo);
100  // ^ Do it yourself and see what the output is
101 
102  // After everything we must free them all
103  PolyDestroy(&p);
104  PolyDestroy(&q);
105  PolyDestroy(&r);
106  PolyDestroy(&sum);
107  PolyDestroy(&rAtTwo);
108 
109  }
110 ```
