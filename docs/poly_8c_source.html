148 /*
149 * Adding two polynomials with optional scaling of second parameter
150 */
151 Poly PolyAddScaled(const Poly *p, const Poly *q, const poly_coeff_t c) {
152  assert(p!=NULL);
153  assert(q!=NULL);
154  List result = ListNew();
155 
156  ListIterator iq = ListBegin(&(q->monos));
157  ListIterator ip = ListBegin(&(p->monos));
158  while(ip != NULL || iq != NULL) {
159  Mono* mp = (Mono*) ListGetValue(ip);
160  Mono* mq = (Mono*) ListGetValue(iq);
161  if(mp == NULL) {
162  Mono* new_mono = MonoClonePtr(mq);
163  PolyScaleConst(&(new_mono->p), c);
164  ListPushBack(&result, new_mono);
165  iq = ListNext(iq);
166  } else if(mq == NULL) {
167  Mono* new_mono = MonoClonePtr(mp);
168  ListPushBack(&result, new_mono);
169  ip = ListNext(ip);
170  } else if(mp->exp > mq->exp) {
171  Mono* new_mono = MonoClonePtr(mq);
172  PolyScaleConst(&(new_mono->p), c);
173  ListPushBack(&result, new_mono);
174  iq = ListNext(iq);
175  } else if(mp->exp < mq->exp) {
176  Mono* new_mono = MonoClonePtr(mp);
177  ListPushBack(&result, new_mono);
178  ip = ListNext(ip);
179  } else {
180  Poly polyAddResult = PolyAddScaled(&(mp->p), &(mq->p), c);
181 
182  if(PolyIsCoeff(&polyAddResult) && PolyGetConstTerm(&polyAddResult) == 0) {
183  PolyDestroy(&polyAddResult);
184  } else {
185  Mono* partial_result = MALLOCATE(Mono);
186  *partial_result = (Mono) { .exp = mp->exp, .p = polyAddResult };
187  ListPushBack(&result, partial_result);
188  }
189  ip = ListNext(ip);
190  iq = ListNext(iq);
191  }
192  }
193  return (Poly) { .c = p->c + (q->c)*c, .monos = result };
194 }

238 * - simplifying (the monomial coefficient was set to zero after addition)

373 /*
374 * Multiply two polynomials
375 */
376 Poly PolyMul(const Poly *p, const Poly *q) {
377  assert(p!=NULL);
378  assert(q!=NULL);
379 
380  Poly result = PolyFromCoeff(0);
381 
382  if(q->c != 0) {
383  LOOP_LIST(&(p->monos), pIter) {
384  Mono* partialResult = MonoClonePtr( (Mono*)ListGetValue(pIter) );
385  PolyScaleConst(&(partialResult->p), q->c);
386  PolyInsertMonoPtr(&result, partialResult);
387  }
388  }
389 
390  if(p->c != 0) {
391  LOOP_LIST(&(q->monos), qIter) {
392  Mono* partialResult = MonoClonePtr( (Mono*)ListGetValue(qIter) );
393  PolyScaleConst(&(partialResult->p), p->c);
394  PolyInsertMonoPtr(&result, partialResult);
395  }
396  }
397 
398  LOOP_LIST(&(p->monos), pIter) {
399  LOOP_LIST(&(q->monos), qIter) {
400  Mono* mp = (Mono*) ListGetValue(pIter);
401  Mono* mq = (Mono*) ListGetValue(qIter);
402  Poly factPartialResult = PolyMul(&(mp->p), &(mq->p));
403  Mono partialResult = MonoFromPoly(&factPartialResult, mp->exp + mq->exp);
404  PolyInsertMonoValue(&result, partialResult);
405  }
406  }
407 
408  result.c += q->c * p->c;
409  return result;
410 }

476 * Recursive equality test for polynomials
477 */
478 bool PolyIsEqRec(const Poly *p, const Poly *q) {
479  if(p == q) return true;
480  if(p->c != q->c) {
481  return false;
482  }
483 
484  ListIterator iq = ListBegin(&(q->monos));
485  LOOP_LIST(&(p->monos), ip) {
486  Mono* mp = (Mono*) ListGetValue(ip);
487  Mono* mq = (Mono*) ListGetValue(iq);
488  if(mp!=mq) {
489  if((ip == NULL && iq != NULL) || (ip != NULL && iq == NULL)) {
490  return false;
491  }
492  if(mp->exp != mq->exp) {
493  return false;
494  }
495  if(!PolyIsEqRec(&(mp->p), &(mq->p))) {
496  return false;
497  }
498  }
499  iq = ListNext(iq);
500  }
501  return true;
502 }

521  if(p->c == 1) return PolyFromCoeff(1);

543 /*
544 * Evaluate polynomial at a given point
545 */
546 Poly PolyAt(const Poly *p, poly_coeff_t x) {
547 
548  assert(p!=NULL);
549 
550  Poly result = PolyFromCoeff(0);
551  result.c += p->c;
552 
553  LOOP_LIST(&(p->monos), i) {
554  Mono* m = (Mono*) ListGetValue(i);
555  poly_coeff_t factValue = MathFastPowLong(x, m->exp);
556 
557  Poly partialResult = PolyClone(&(m->p));
558  PolyScaleConst(&partialResult, factValue);
559  result.c += partialResult.c;
560  partialResult.c = 0;
561  LOOP_LIST(&(partialResult.monos), j) {
562  Mono* submono = (Mono*) ListGetValue(j);
563  Poly pcln = PolyClone(&(submono->p));
564  Mono cln = MonoFromPoly(&pcln, submono->exp);
565 
566  PolyInsertMonoValue(&result, cln);
567  }
568  PolyDestroy(&partialResult);
569 
570  }
571 
572  return result;
573 }
