645 static inline Poly PolyBuild(int dummy, ...) {
646  va_list list;
647  va_start(list, dummy);
648  unsigned count = 0;
649  while (true) {
650  va_arg(list, Poly);
651  if (va_arg(list, int) < 0)
652  break;
653  count++;
654  }
655  va_start(list, dummy);
656  Mono *arr = calloc(count, sizeof(Mono));
657  for (unsigned i = 0; i < count; ++i) {
658  Poly p = va_arg(list, Poly);
659  arr[i] = MonoFromPoly(&p, va_arg(list, int));
660  assert(i == 0 || arr[i].exp > arr[i - 1].exp);
661  }
662  Poly closing_zero = va_arg(list, Poly);
663  va_end(list);
664  PolyDestroy(&closing_zero);
665  Poly res = PolyAddMonos(count, arr);
666  free(arr);
667  return res;
668 }
